Leetcode Problem 103: Binary Tree Zigzag Level Order Traversal

Problem Link:

Explanation:


 Java Solution: 
 
     n = # elements in binary Tree
     BFS(Bredth First Search): Space: O(n) Time: O(n)

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<List<Integer>> zigzagLevelOrder(TreeNode root) {
        
        List<List<Integer>> returnarr = new ArrayList<>();
        Queue<TreeNode> rowQueue = new LinkedList<>();

        if(root == null){ return returnarr; }  
        rowQueue.add(root);
        boolean reverseRow = false;
        TreeNode curr;

        while(!rowQueue.isEmpty()){
            int rowlen = rowQueue.size();
            List<Integer> currRow = new ArrayList<>();
            
            for(int i = 0; i < rowlen; i++){
                curr = rowQueue.poll();
                currRow.add(curr.val);
                if(curr.left != null){ rowQueue.add(curr.left); }
                if(curr.right != null){ rowQueue.add(curr.right); }
            }
            if(reverseRow){ Collections.reverse(currRow);}
            reverseRow = !reverseRow;
            returnarr.add(currRow);
        }
        return returnarr;
    }
}
```
